// Hello World 示例
use <stdio.emh>

    println("Hello, Emerging World!");
    
    int x = 42;
    printf("The answer is: %d\n", x);
    
    ret 0;
}
// 斐波那契数列
use <stdio.emh>

int fib(int n) {
    if (n <= 1) {
        ret n;
    }
    ret fib(n - 1) + fib(n - 2);
}

int main() {
    for (int i = 0; i < 10; i++) {
        printf("fib(%d) = %d\n", i, fib(i));
    }
    ret 0;
}
// 结构体定义
struct Point {
    int x;
    int y;
};

struct Rectangle {
    struct Point top_left;
    struct Point bottom_right;
};

int main() {
    struct Point p = {10, 20};
    printf("Point: (%d, %d)\n", p.x, p.y);
    ret 0;
}
// 内存管理
use <stdio.emh>
use <memory.emh>

int main() {
    // 动态分配内存
    int* arr = (int*)malloc(10 * sizeof(int));
    
    for (int i = 0; i < 10; i++) {
        arr[i] = i * i;
    }
    
    free(arr);
    ret 0;
}
